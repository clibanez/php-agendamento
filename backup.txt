// Processar exclusão de serviço
if (isset($_POST['excluir_servico'])) {
    $servico_id = $_POST['servico_id'];
    try {
        // Buscar a imagem do serviço antes de excluir
        $stmt = $pdo->prepare("SELECT imagem_url FROM servicos WHERE id = ?");
        $stmt->execute([$servico_id]);
        $imagem = $stmt->fetchColumn();

        // Excluir o serviço do banco de dados
        $stmt = $pdo->prepare("DELETE FROM servicos WHERE id = ?");
        $stmt->execute([$servico_id]);

        // Remover a imagem do servidor, se existir
        if ($imagem && file_exists($imagem)) {
            unlink($imagem);
        }

        header("Location: servicos.php");
        exit();
    } catch (PDOException $e) {
        error_log("Erro ao excluir serviço: " . $e->getMessage());
        $erro = "Erro ao excluir serviço: " . $e->getMessage();
    }
}

    $stmt = $pdo->query("SELECT * FROM servicos");

    error_log("Erro ao buscar serviços: " . $e->getMessage());

<?php
session_start();
require_once '../config.php';

// Verificar se o usuário está logado e é um administrador
if (!isset($_SESSION['usuario_id']) || $_SESSION['usuario_tipo'] !== 'admin') {
    header("Location: ../index.php");
    exit();
}

// Processar o formulário de cadastro
if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    $nome = $_POST['nome'] ?? '';
    $descricao = $_POST['descricao'] ?? '';
    $preco = $_POST['preco'] ?? '';
    $status = $_POST['status'] ?? 'ativo';
    $duracao_minutos = $_POST['duracao_minutos'] ?? '';
    $empresa_id = $_POST['empresa_id'] ?? '';
    $imagem = $_FILES['imagem'] ?? null;

    // Validação básica
    if (empty($nome) || empty($descricao) || empty($preco) || empty($duracao_minutos) || empty($empresa_id) || !$imagem) {
        $erro = "Todos os campos são obrigatórios.";
    } else {
        // Obter o nome da empresa do usuário logado
        $empresa_nome = $_SESSION['empresa_nome'];

        // Criar diretório uploads se não existir
        $uploads_dir = "../uploads/";
        if (!file_exists($uploads_dir)) {
            mkdir($uploads_dir, 0755, true);
        }
        // Criar diretório da empresa se não existir
        $target_dir = $uploads_dir . "empresas/" . $empresa_nome . "/servicos-fotos/";
        if (!file_exists($target_dir)) {
            mkdir($target_dir, 0755, true);
        }

        // Definir o caminho do arquivo
        $target_file = $target_dir . basename($imagem['name']);

        // Verificar se o upload foi bem-sucedido
        if (move_uploaded_file($imagem['tmp_name'], $target_file)) {
            try {
                $stmt = $pdo->prepare("INSERT INTO servicos (nome, descricao, preco, status, duracao_minutos, empresa_id, imagem_url) VALUES (?, ?, ?, ?, ?, ?, ?)");
                $stmt->execute([$nome, $descricao, $preco, $status, $duracao_minutos, $empresa_id, $target_file]);
                header("Location: servicos.php");
                exit();
            } catch (PDOException $e) {
                error_log("Erro ao cadastrar serviço: " . $e->getMessage());
                $erro = "Erro ao cadastrar serviço: " . $e->getMessage();
            }
        } else {
            $erro = "Erro ao fazer upload da imagem.";
        }
    }
}
